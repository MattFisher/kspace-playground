# Undo/Redo Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add undo/redo to the k-space Playground — one stroke/action per step, via canonical-buffer snapshots — with toolbar buttons and keyboard shortcuts.

**Architecture:** A new pure `src/engine/history.ts` module owns two snapshot stacks and wraps the store through its public interface. A snapshot copies only the last-edited (canonical) buffer; restoring recomputes the derived side with one transform. The interaction layer marks action boundaries (`begin`/`commit`/`cancel`) so a coalesced drag is one step; imports, clear, and examples each wrap their mutation the same way.

**Tech Stack:** TypeScript, Vite, Vitest. No new dependencies.

**Spec:** `docs/superpowers/specs/2026-06-04-editing-controls-design.md` (§4)

**Prerequisite:** Assumes the Clear button and Examples feature from
`docs/superpowers/plans/2026-06-04-clear-and-examples.md` are already implemented (this plan
wraps their handlers with history boundaries).

**Conventions:** Buffers are row-major `Float32Array` length `N*N`. `Spectrum` is
`{ re: Float32Array, im: Float32Array }` (from `src/engine/fft2d.ts`). The store
(`src/engine/state.ts`) exposes `N`, `getSpatial`, `getSpectrum`, `lastEdited`,
`setSpatial`, `setSpectrum`, and mutates buffers in place — so snapshots must deep-copy.

```ts
    store.editSpatial(brushCoverage(4, 1, 1, 1, 0), 1);

    history.undo();
    history.redo();

    history.commit();
    history.undo();
    expect(history.canRedo()).toBe(true);
    history.begin();
    store.editSpatial(brushCoverage(4, 2, 2, 1, 0), 1);
    history.commit();
    expect(history.canRedo()).toBe(false);
  });

  it("cancel discards the pending baseline (no step added)", () => {
    const { history } = setup();
    history.begin();
    history.cancel();
    expect(history.canUndo()).toBe(false);

      store.editSpatial(brushCoverage(4, i % 4, 0, 1, 0), 1);
      history.commit();
    }
    let undos = 0;
    while (history.canUndo()) {
      history.undo();
      undos++;
    }
    expect(undos).toBe(3);
  });
```

```ts
// src/engine/history.ts
import type { Spectrum } from "./fft2d";
import type { Store } from "./state";

type Snapshot =
  | { domain: "spatial"; buffer: Float32Array }
  | { domain: "kspace"; buffer: Spectrum }
  | { domain: null; buffer: null };

export interface History {
  /** Capture the current state as the pre-action baseline (idempotent per action). */
  begin(): void;
  /** An edit occurred: push the baseline to the undo stack and clear redo. */
  commit(): void;
  /** No edit occurred: discard the baseline. */
  cancel(): void;
  undo(): void;
  redo(): void;
  canUndo(): boolean;
  canRedo(): boolean;
  /** Notified after any stack change (for button enable/disable). */
  subscribe(fn: () => void): () => void;
}

function snapshot(store: Store): Snapshot {
  const domain = store.lastEdited();
  if (domain === "spatial") {
    return { domain, buffer: Float32Array.from(store.getSpatial()) };
  }
  if (domain === "kspace") {
    const s = store.getSpectrum();
    return {
      domain,
      buffer: { re: Float32Array.from(s.re), im: Float32Array.from(s.im) },
    };
  }
  return { domain: null, buffer: null };
}

function restore(store: Store, snap: Snapshot): void {
  if (snap.domain === "spatial") {
    store.setSpatial(Float32Array.from(snap.buffer));
  } else if (snap.domain === "kspace") {
    store.setSpectrum({
      re: Float32Array.from(snap.buffer.re),
      im: Float32Array.from(snap.buffer.im),
    });
  } else {
    store.setSpatial(new Float32Array(store.N * store.N));
  }
}

export function createHistory(store: Store, opts: { maxDepth: number }): History {
  const undoStack: Snapshot[] = [];
  const redoStack: Snapshot[] = [];
  let pending: Snapshot | null = null;
  const listeners = new Set<() => void>();
  const emit = () => listeners.forEach((fn) => fn());

  return {
    begin() {
      if (pending === null) pending = snapshot(store);
    },
    commit() {
      if (pending === null) return;
      undoStack.push(pending);
      if (undoStack.length > opts.maxDepth) undoStack.shift();
      pending = null;
      redoStack.length = 0;
      emit();
    },
    cancel() {
      pending = null;
    },
    undo() {
      if (undoStack.length === 0) return;
      redoStack.push(snapshot(store));
      restore(store, undoStack.pop()!);
      emit();
    },
    redo() {
      if (redoStack.length === 0) return;
      undoStack.push(snapshot(store));
      restore(store, redoStack.pop()!);
      emit();
    },
    canUndo: () => undoStack.length > 0,
    canRedo: () => redoStack.length > 0,
    subscribe(fn) {
      listeners.add(fn);
      return () => listeners.delete(fn);
    },
  };
}
```

```ts
const undoBtn = $("undo") as HTMLButtonElement;
const redoBtn = $("redo") as HTMLButtonElement;

function refreshHistoryButtons() {
  undoBtn.disabled = !history.canUndo();
  redoBtn.disabled = !history.canRedo();
}
history.subscribe(refreshHistoryButtons);
refreshHistoryButtons();

undoBtn.addEventListener("click", () => history.undo());
redoBtn.addEventListener("click", () => history.redo());

window.addEventListener("keydown", (e) => {
  const target = e.target as HTMLElement;
  // Ignore shortcuts while typing in inputs.
  if (target instanceof HTMLInputElement || target instanceof HTMLSelectElement) return;
  const mod = e.metaKey || e.ctrlKey;
  if (!mod) return;
  const key = e.key.toLowerCase();
  if (key === "z" && !e.shiftKey) {
    e.preventDefault();
    history.undo();
  } else if ((key === "z" && e.shiftKey) || key === "y") {
    e.preventDefault();
    history.redo();
  }
});
```

## Notes for the implementer

- The `edited` flag is set synchronously in `queueEdit` even though the brush transform
  runs on the next animation frame. That is correct: the undo baseline is captured at
  `pointerdown` (pre-stroke), so committing on `pointerup` records the right step regardless
  of when the queued transform actually applies.
- `restore` of the pristine (`domain: null`) snapshot uses `setSpatial(zeros)`, which leaves
  `lastEdited() === "spatial"`. That is functionally identical to the pristine state, so
  redo after undoing to pristine still behaves correctly.
- Colormap and tool/parameter changes are intentionally NOT undoable — they never mutate the
  canonical buffers.
- Memory: undo depth is capped at 30 snapshots; worst case ≈ 30 × 2 MB (k-space) = 60 MB.
